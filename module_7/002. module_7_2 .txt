def custom_write(file_name, strings):

    file = open(file_name, "w", encoding="utf-8")

    result = {}
    n_s = 0
    for i in strings:
        n_s += 1

        poz = file.tell()

        file.write(i+"\n")

        result[(n_s, poz)]= i

    file.close()

    return result

info = [

    'Text for tell.',

    'Используйте кодировку utf-8.',

    'Because there are 2 languages!',

    'Спасибо!'

    ]



result = custom_write('test.txt', info)

for elem in result.items():

  print(elem)
